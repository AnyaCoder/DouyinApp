# 创建expo app仿短视频应用
npx create-expo-app DouyinApp

# 安装资源
npm install @react-navigation/native
npm install eact-native-reanimated 
npx expo install react-native-gesture-handler 
npm install react-native-screens@3.29.0
npm install react-native-safe-area-context@4.8.2
npm install @react-native-community/masked-view

# 安装导航器
npm install @react-navigation/stack
npm install @react-navigation/bottom-tabs
npm install react-native-vector-icons

# 安装 ViewPager，可以实现屏幕滑动的库。
npm install @react-native-community/viewpager

# 安装拍照录像包
npm install react-native-image-crop-picker
# 安装视频播放器
npm install react-native-video
# 安装tabview
npm install react-native-scrollable-tab-view

# 在首页（index.js or App.js）的第一行添加
# import  'react-native-gesture-handler';

# 运行
npm run android
